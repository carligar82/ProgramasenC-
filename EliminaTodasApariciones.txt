#include <iostream>
#include <string>
using namespace std;

void EliminaTodasApariciones (string &frase , string pal)
{
 int i = frase.find (pal);
 while (i >= 0)
 {
  frase.erase (i , pal.length());
  i = frase.find (pal);
 }
 /*return;*/
}

{
    string frase="casa",pal="a";
